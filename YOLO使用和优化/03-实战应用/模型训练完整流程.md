```python
"""
YOLOv8 完整训练流程示例
功能: 从数据加载到模型导出的端到端训练
"""

import os
import sys
from pathlib import Path
from datetime import datetime
import torch
import yaml
from ultralytics import YOLO


class YOLOTrainingPipeline:
    """YOLO训练管道类"""
    
    def __init__(self, 
                 project_name='my_yolo_project',
                 base_dir=None):
        """
        初始化训练管道
        
        参数:
            project_name: 项目名称
            base_dir: 基础目录 (默认为当前目录)
        """
        if base_dir is None:
            base_dir = Path.cwd()
        
        self.base_dir = Path(base_dir)
        self.project_name = project_name
        self.timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        
        # 目录设置
        self.data_dir = self.base_dir / 'data'
        self.configs_dir = self.base_dir / 'configs'
        self.runs_dir = self.base_dir / 'runs'
        self.output_dir = self.runs_dir / f'{project_name}_{self.timestamp}'
        
        # 创建必要目录
        for d in [self.configs_dir, self.output_dir]:
            d.mkdir(parents=True, exist_ok=True)
    
    def check_environment(self):
        """检查运行环境"""
        print("\n" + "="*60)
        print("🔍 环境检查")
        print("="*60)
        
        # Python版本
        print(f"✅ Python版本: {sys.version.split()[0]}")
        
        # PyTorch和CUDA
        print(f"✅ PyTorch版本: {torch.__version__}")
        if torch.cuda.is_available():
            print(f"✅ CUDA可用: 是")
            print(f"✅ GPU设备: {torch.cuda.get_device_name(0)}")
            print(f"✅ 显存大小: {torch.cuda.get_device_properties(0).total_mem / 1024**3:.1f} GB")
            self.device = 0
        else:
            print("⚠️  CUDA不可用，将使用CPU")
            self.device = 'cpu'
        
        # 检查ultralytics
        try:
            import ultralytics
            print(f"✅ Ultralytics版本: {ultralytics.__version__}")
        except ImportError:
            raise ImportError("请先安装ultralytics: pip install ultralytics")
        
        return True
    
    def prepare_dataset(self, dataset_yaml_path):
        """
        验证并准备数据集
        
        参数:
            dataset_yaml_path: 数据集YAML配置文件路径
        """
        print("\n" + "="*60)
        print("📦 数据集准备")
        print("="*60)
        
        yaml_path = Path(dataset_yaml_path)
        
        if not yaml_path.exists():
            raise FileNotFoundError(f"数据集配置文件不存在: {yaml_path}")
        
        # 加载YAML配置
        with open(yaml_path, 'r', encoding='utf-8') as f:
            config = yaml.safe_load(f)
        
        print(f"\n📋 数据集配置:")
        print(f"   路径: {config.get('path', 'N/A')}")
        print(f"   训练集: {config.get('train', 'N/A')}")
        print(f"   验证集: {config.get('val', 'N/A')}")
        print(f"   类别数: {config.get('nc', 0)}")
        print(f"   类别名: {list(config.get('names', {}).values())}")
        
        # 检查数据集目录
        base_path = self.base_dir / config['path']
        
        required_dirs = [
            base_path / config.get('train', '').replace('images/', ''),
            base_path / config.get('val', '').replace('images/', '')
        ]
        
        for dir_path in required_dirs:
            if dir_path.exists():
                num_files = len(list(dir_path.glob('*')))
                print(f"   ✅ {dir_path.name}: {num_files} 个文件")
            else:
                print(f"   ❌ {dir_path}: 不存在!")
        
        self.dataset_yaml = str(yaml_path.resolve())
        return self.dataset_yaml
    
    def configure_training(self,
                          model_type='yolov8n',
                          epochs=100,
                          batch_size=16,
                          image_size=640,
                          **kwargs):
        """
        配置训练参数
        
        参数:
            model_type: 模型类型 (yolov8n/s/m/l/x)
            epochs: 训练轮数
            batch_size: 批量大小
            image_size: 输入图像尺寸
            **kwargs: 其他训练参数
        """
        print("\n" + "="*60)
        print("⚙️  训练配置")
        print("="*60)
        
        self.model_type = model_type
        self.training_config = {
            # 基本参数
            'data': self.dataset_yaml,
            'epochs': epochs,
            'batch': batch_size,
            'imgsz': image_size,
            
            # 模型设置
            'model': f'{model_type}.pt' if model_type.endswith('.pt') else model_type,
            
            # 硬件
            'device': self.device,
            'workers': 4 if torch.cuda.is_available() else 2,
            
            # 性能优化
            'amp': True,  # 自动混合精度
            
            # 输出设置
            'project': str(self.output_dir),
            'name': 'training',
            'exist_ok': False,
            'verbose': True,
            
            # 可复现性
            'seed': 42,
            'deterministic': True,
            
            # 更新其他参数
            **kwargs
        }
        
        print(f"\n📊 训练参数:")
        for key, value in self.training_config.items():
            if key != 'data':
                print(f"   {key}: {value}")
        
        return self.training_config
    
    def run_training(self):
        """执行训练"""
        print("\n" + "="*60)
        print("🚀 开始训练")
        print("="*60 + "\n")
        
        # 加载模型
        print(f"📥 加载模型: {self.training_config['model']}")
        self.model = YOLO(self.training_config['model'])
        
        # 开始训练
        print(f"⏱️  训练参数:")
        print(f"   - Epochs: {self.training_config['epochs']}")
        print(f"   - Batch size: {self.training_config['batch']}")
        print(f"   - Image size: {self.training_config['imgsz']}")
        print(f"   - Device: {self.device}\n")
        
        # 执行训练
        self.results = self.model.train(**self.training_config)
        
        return self.results
    
    def evaluate_model(self, conf_threshold=0.25, iou_threshold=0.65):
        """评估训练好的模型"""
        print("\n" + "="*60)
        print("✅ 模型评估")
        print("="*60)
        
        # 加载最佳模型
        best_model_path = self.results.save_dir / 'weights' / 'best.pt'
        
        if best_model_path.exists():
            print(f"\n📥 加载最佳模型: {best_model_path}")
            eval_model = YOLO(str(best_model_path))
        else:
            print("\n⚠️  未找到最佳模型，使用最后模型")
            last_model_path = self.results.save_dir / 'weights' / 'last.pt'
            eval_model = YOLO(str(last_model_path))
        
        # 运行验证
        metrics = eval_model.val(
            data=self.dataset_yaml,
            conf=conf_threshold,
            iou=iou_threshold,
            batch=16,
            imgsz=self.training_config['imgsz'],
            plots=True,
            verbose=True
        )
        
        # 打印关键指标
        print(f"\n📈 评估结果:")
        print(f"   mAP@0.5:       {metrics.box.map50:.4f}")
        print(f"   mAP@0.5:0.95:  {metrics.box.map:.4f}")
        print(f"   mAP@0.75:       {metrics.box.map75:.4f}")
        print(f"   Precision:     {metrics.box.mp:.4f}")
        print(f"   Recall:        {metrics.box.mr:.4f}")
        
        # 保存评估报告
        report_path = self.output_dir / 'evaluation_report.txt'
        with open(report_path, 'w', encoding='utf-8') as f:
            f.write(f"评估时间: {datetime.now()}\n\n")
            f.write(f"模型: {best_model_path}\n\n")
            f.write(f"性能指标:\n")
            f.write(f"  mAP@0.5:      {metrics.box.map50:.4f}\n")
            f.write(f"  mAP@0.5:0.95: {metrics.box.map:.4f}\n")
            f.write(f"  mAP@0.75:      {metrics.box.map75:.4f}\n")
            f.write(f"  Precision:    {metrics.box.mp:.4f}\n")
            f.write(f"  Recall:       {metrics.box.mr:.4f}\n")
        
        print(f"\n💾 评估报告已保存: {report_path}")
        
        return metrics
    
    def export_model(self, formats=['onnx']):
        """导出模型到不同格式"""
        print("\n" + "="*60)
        print("📤 模型导出")
        print("="*60)
        
        # 加载最佳模型
        best_model_path = self.results.save_dir / 'weights' / 'best.pt'
        export_model = YOLO(str(best_model_path))
        
        export_results = {}
        
        for fmt in formats:
            try:
                print(f"\n🔄 导出格式: {fmt.upper()}")
                
                export_path = export_model.export(
                    format=fmt,
                    imgsz=self.training_config['imgsz'],
                    half=True if fmt == 'engine' else False,  # TensorRT用FP16
                    simplify=True if fmt == 'onnx' else False,
                    workspace=4 if fmt == 'engine' else 1,
                )
                
                export_results[fmt] = {
                    'success': True,
                    'path': export_path
                }
                print(f"   ✅ 导出成功: {export_path}")
                
            except Exception as e:
                export_results[fmt] = {
                    'success': False,
                    'error': str(e)
                }
                print(f"   ❌ 导出失败: {str(e)[:50]}")
        
        # 保存导出日志
        log_path = self.output_dir / 'export_log.yaml'
        with open(log_path, 'w', encoding='utf-8') as f:
            yaml.dump(export_results, f, allow_unicode=True)
        
        return export_results


def main():
    """主函数：完整的训练流程演示"""
    
    print("╔" + "═"*58 + "╗")
    print("║" + " "*15 + "YOLOv8 完整训练流程" + " "*18 + "║")
    print("╚" + "═"*58 + "╝")
    
    # ===== 初始化 =====
    pipeline = YOLOTrainingPipeline(
        project_name='pet_detection',
        base_dir=Path(__file__).parent.parent
    )
    
    # ===== 1. 环境检查 =====
    pipeline.check_environment()
    
    # ===== 2. 准备数据集 =====
    dataset_yaml = pipeline.prepare_dataset('data/data.yaml')
    
    # ===== 3. 配置训练 =====
    pipeline.configure_training(
        model_type='yolov8n',       # 使用nano版快速测试
        epochs=100,                  # 训练轮数
        batch_size=16,               # 批量大小
        image_size=640,              # 图像尺寸
        
        # 高级参数（可选）
        optimizer='SGD',
        lr0=0.01,
        lrf=0.01,
        patience=20,                # 早停耐心值
        
        # 数据增强
        hsv_h=0.015,
        hsv_s=0.7,
        hsv_v=0.4,
        degrees=10.0,
        translate=0.1,
        scale=0.5,
        fliplr=0.5,
        mosaic=1.0,
    )
    
    # ===== 4. 执行训练 =====
    results = pipeline.run_training()
    
    # ===== 5. 评估模型 =====
    metrics = pipeline.evaluate_model(conf_threshold=0.25, iou_threshold=0.65)
    
    # ===== 6. 导出模型 =====
    export_results = pipeline.export_model(formats=['onnx', 'engine'])
    
    # ===== 完成! =====
    print("\n" + "="*60)
    print("🎉 训练流程全部完成!")
    print("="*60)
    print(f"\n📁 输出目录: {pipeline.output_dir}")
    print(f"\n生成的文件:")
    print(f"  📊 训练结果: runs/detect/training/")
    print(f"  🏆 最佳模型: runs/detect/training/weights/best.pt")
    print(f"  📈 评估图表: runs/detect/val/")
    print(f"  📤 导出模型: {pipeline.output_dir}/")
    
    return results, metrics


if __name__ == '__main__':
    results, metrics = main()
```

```python
def resume_training(last_pt_path, additional_epochs=50):
    """
    从断点恢复训练
    
    参数:
        last_pt_path: 上次训练的last.pt权重路径
        additional_epochs: 额外训练的轮数
    """
    from ultralytics import YOLO
    
    # 加载最后的检查点
    model = YOLO(last_pt_path)
    
    # 继续训练 (自动从上次epoch继续)
    results = model.train(
        resume=True,                    # 关键参数!
        epochs=model.epochs + additional_epochs,  # 新的总轮数
    )
    
    return results


# 使用示例
resume_training('runs/detect/train/weights/last.pt', additional_epochs=50)
```

```python
def distributed_training(data_yaml, gpus=[0, 1]):
    """多GPU训练"""
    from ultralytics import YOLO
    
    model = YOLO('yolov8n.pt')
    
    results = model.train(
        data=data_yaml,
        epochs=100,
        batch=64,                       # 总batch (每个GPU 32)
        device=gpus,                    # [0, 1] 使用GPU 0和1
        workers=8,
        amp=True,
    )
    
    return results
```

### 冻结层训练 (迁移学习)

```python
def freeze_backbone_training(data_yaml, freeze_layers=10):
    """
    冻结Backbone的前N层进行微调
    
    适用场景:
    - 小数据集 (<1000张)
    - 快速适应新领域
    - 防止破坏预训练特征
    """
    from ultralytics import YOLO
    import torch.nn as nn
    
    model = YOLO('yolov8n.pt')
    
    # 获取模型内部结构
    internal_model = model.model
    
    # 冻结前N个模块
    frozen_count = 0
    for i, module in enumerate(internal_model.model):
        if i < freeze_layers:
            for param in module.parameters():
                param.requires_grad = False
            frozen_count += 1
            print(f"❄️  冻结层 {i}: {module.__class__.__name__}")
    
    print(f"\n总共冻结了 {frozen_count} 层")
    
    # 开始微调 (使用较低的学习率)
    results = model.train(
        data=data_yaml,
        epochs=50,
        batch=16,
        lr0=0.001,                    # 降低学习率
        patience=30,
        freeze=freeze_layers,         # Ultralytics内置支持!
    )
    
    return results
```

```python
def setup_tensorboard_logging(log_dir='runs/tensorboard'):
    """配置TensorBoard日志记录"""
    from torch.utils.tensorboard import SummaryWriter
    
    writer = SummaryWriter(log_dir)
    
    # 在训练回调中使用
    def on_train_epoch_end(trainer):
        writer.add_scalar('Loss/train', trainer.loss_items, trainer.epoch)
        writer.add_scalar('LR', trainer.optimizer.param_groups[0]['lr'], trainer.epoch)
        writer.flush()
    
    return writer, on_train_epoch_end


# Weights & Biases (W&B) 集成
def setup_wandb_logging(project_name='yolo-experiments'):
    """配置W&B实验跟踪"""
    import wandb
    
    wandb.init(
        project=project_name,
        config={
            'learning_rate': 0.01,
            'epochs': 100,
            'batch_size': 16,
        }
    )
    
    def wandb_callback(trainer):
        wandb.log({
            'train_loss': trainer.loss_items,
            'lr': trainer.optimizer.param_groups[0]['lr'],
            'epoch': trainer.epoch
        })
    
    return wandb_callback
```

## 💡 常见问题快速解决

| 问题 | 原因 | 解决方案 |
|------|------|----------|
| Loss不下降 | 学习率过高或数据问题 | 降低lr0，检查数据质量 |
| 过拟合 | 数据太少或模型太大 | 增加数据，使用更强增强 |
| OOM错误 | Batch过大或图像尺寸大 | 减小batch或imgsz |
| 训练太慢 | CPU瓶颈或数据加载慢 | 增加workers，使用cache |
