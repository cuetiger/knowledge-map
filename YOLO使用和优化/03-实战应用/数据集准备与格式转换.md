## 📁 YOLO数据集格式规范

### 目录结构

```
dataset/
├── data.yaml              # 数据集配置文件
├── images/
│   ├── train/             # 训练图像
│   │   ├── image001.jpg
│   │   ├── image002.jpg
│   │   └── ...
│   └── val/               # 验证图像
│       ├── image101.jpg
│       ├── image102.jpg
│       └── ...
└── labels/
    ├── train/             # 训练标签
    │   ├── image001.txt
    │   ├── image002.txt
    │   └── ...
    └── val/               # 验证标签
        ├── image101.txt
        ├── image102.txt
        └── ...
```

### 标注文件格式 (.txt)

**每行格式**: `class_id x_center y_center width height`

```text
# 示例: image001.txt (假设图像尺寸为640x480)
0 0.500000 0.625000 0.250000 0.583333    # person, 中心在(320,300), 宽160, 高280
2 0.750000 0.520833 0.375000 0.458333    # car, 中心在(480,250), 宽240, 高220
16 0.125000 0.708333 0.156250 0.333333  # dog, 中心在(80,340), 宽100, 高160
```

**重要说明**:
- 所有坐标值都是 **归一化到 [0, 1]** 范围的浮点数
- 坐标原点在 **左上角**
- `x_center`, `y_center` 是边界框中心的相对位置
- `width`, `height` 是边界框宽高的相对尺寸

```python
import json
import os


def labelme_to_yolo(labelme_json_path, output_txt_path, img_width, img_height):
    """
    将LabelMe的JSON格式转换为YOLO格式
    
    参数:
        labelme_json_path: LabelMe导出的JSON文件路径
        output_txt_path: 输出的YOLO格式txt文件路径
        img_width: 图像宽度
        img_height: 图像高度
    """
    with open(labelme_json_path, 'r') as f:
        data = json.load(f)
    
    yolo_labels = []
    
    for shape in data['shapes']:
        class_id = shape['label']  # 类别名称或ID
        points = shape['points']   # 多边形顶点
        
        if len(points) >= 4:  # 至少需要4个点形成多边形
            # 计算外接矩形
            xs = [p[0] for p in points]
            ys = [p[1] for p in points]
            
            x_min = min(xs)
            y_min = min(ys)
            x_max = max(xs)
            y_max = max(ys)
            
            # 转换为YOLO格式 (归一化)
            x_center = ((x_min + x_max) / 2) / img_width
            y_center = ((y_min + y_max) / 2) / img_height
            width = (x_max - x_min) / img_width
            height = (y_max - y_min) / img_height
            
            yolo_label = f"{class_id} {x_center:.6f} {y_center:.6f} {width:.6f} {height:.6f}"
            yolo_labels.append(yolo_label)
    
    with open(output_txt_path, 'w') as f:
        f.write('\n'.join(yolo_labels))
```

### VOC → YOLO 转换

import xml.etree.ElementTree as ET
from pathlib import Path
import os


def voc_to_yolo(voc_xml_dir, yolo_label_dir, class_names, output_img_size=(640, 640)):
    """
    将VOC XML格式转换为YOLO txt格式
    
    参数:
        voc_xml_dir: VOC XML文件目录
        yolo_label_dir: 输出YOLO标签目录
        class_names: 类别名称列表 ['person', 'car', ...]
        output_img_size: 输出图像尺寸 (用于归一化)
    
    使用示例:
        >>> class_names = ['person', 'bicycle', 'car', 'motorcycle']
        >>> voc_to_yolo('voc_annotations/', 'yolo_labels/', class_names)
    """
    os.makedirs(yolo_label_dir, exist_ok=True)
    
    # 创建类别名到ID的映射
    class_to_id = {name: idx for idx, name in enumerate(class_names)}
    
    xml_files = list(Path(voc_xml_dir).glob('*.xml'))
    
    print(f"📂 发现 {len(xml_files)} 个XML文件")
    
    converted_count = 0
    
    for xml_file in xml_files:
        try:
            tree = ET.parse(xml_file)
            root = tree.getroot()
            
            # 获取图像尺寸
            size = root.find('size')
            if size is not None:
                img_w = int(size.find('width').text)
                img_h = int(size.find('height').text)
            else:
                img_w, img_h = output_img_size
            
            # 提取所有目标
            yolo_lines = []
            
            for obj in root.findall('object'):
                class_name = obj.find('name').text
                
                # 检查类别是否有效
                if class_name not in class_to_id:
                    print(f"⚠️  未知类别 '{class_name}' 在文件 {xml_file.name}")
                    continue
                
                class_id = class_to_id[class_name]
                
                # 获取边界框坐标 (VOC格式: xmin, ymin, xmax, ymax)
                bbox = obj.find('bndbox')
                xmin = float(bbox.find('xmin').text)
                ymin = float(bbox.find('ymin').text)
                xmax = float(bbox.find('xmax').text)
                ymax = float(bbox.find('ymax').text)
                
                # 转换为YOLO格式 (归一化的中心点+宽高)
                x_center = ((xmin + xmax) / 2) / img_w
                y_center = ((ymin + ymax) / 2) / img_h
                width = (xmax - xmin) / img_w
                height = (ymax - ymin) / img_h
                
                # 确保数值在[0,1]范围内
                x_center = max(0.0, min(1.0, x_center))
                y_center = max(0.0, min(1.0, y_center))
                width = max(0.0, min(1.0, width))
                height = max(0.0, min(1.0, height))
                
                yolo_line = f"{class_id} {x_center:.6f} {y_center:.6f} {width:.6f} {height:.6f}"
                yolo_lines.append(yolo_line)
            
            # 写入YOLO标签文件
            output_file = Path(yolo_label_dir) / (xml_file.stem + '.txt')
            with open(output_file, 'w') as f:
                f.write('\n'.join(yolo_lines))
            
            converted_count += 1
            
        except Exception as e:
            print(f"❌ 处理失败: {xml_file.name} - {str(e)}")
    
    print(f"✅ 成功转换 {converted_count}/{len(xml_files)} 个文件")
    print(f"📁 YOLO标签已保存至: {yolo_label_dir}")


if __name__ == '__main__':

### COCO → YOLO 转换

```python
from pycocotools.coco import COCO
from pathlib import Path
import os


def coco_to_yolo(coco_json_path, output_dir, img_dir=None):
    """
    将COCO JSON格式转换为YOLO格式
    
    参数:
        coco_json_path: COCO annotations JSON文件路径
        output_dir: 输出目录 (包含images/和labels/)
        img_dir: 图像目录 (可选，用于验证)
    
    COCO格式说明:
        annotations.json 包含:
        - images: [{id, file_name, width, height}, ...]
        - annotations: [{id, image_id, category_id, bbox}, ...]
          bbox格式: [x, y, width, height] (未归一化)
        - categories: [{id, name}, ...]
    """
    print("📖 加载COCO数据...")
    coco = COCO(coco_json_path)
    
    # 创建输出目录
    images_out = Path(output_dir) / 'images'
    labels_out = Path(output_dir) / 'labels'
    images_out.mkdir(parents=True, exist_ok=True)
    labels_out.mkdir(parents=True, exist_ok=True)
    
    # 获取所有图像ID
    image_ids = coco.getImgIds()
    print(f"📷 总计 {len(image_ids)} 张图像")
    
    # 类别映射 (COCO ID → 连续ID)
    cat_ids = coco.getCatIds()
    categories = coco.loadCats(cat_ids)
    coco_id_to_continuous = {cat['id']: idx for idx, cat in enumerate(categories)}
    
    converted = 0
    
    for img_id in image_ids:
        # 获取图像信息
        img_info = coco.loadImgs(img_id)[0]
        img_filename = img_info['file_name']
        img_w = img_info['width']
        img_h = img_info['height']
        
        # 获取该图像的所有标注
        ann_ids = coco.getAnnIds(imgIds=img_id)
        anns = coco.loadAnns(ann_ids)
        
        # 转换标注
        yolo_lines = []
        
        for ann in anns:
            # 获取连续类别ID
            cat_id = ann['category_id']
            if cat_id not in coco_id_to_continuous:
                continue
            
            cls_id = coco_id_to_continuous[cat_id]
            
            # COCO bbox: [x, y, w, h] (左上角坐标 + 宽高)
            x, y, w, h = ann['bbox']
            
            # 转换为YOLO格式 (归一化中心坐标 + 宽高)
            x_center = (x + w/2) / img_w
            y_center = (y + h/2) / img_h
            norm_w = w / img_w
            norm_h = h / img_h
            
            # 数值裁剪到[0,1]
            x_center = max(0.0, min(1.0, x_center))
            y_center = max(0.0, min(1.0, y_center))
            norm_w = max(0.0, min(1.0, norm_w))
            norm_h = max(0.0, min(1.0, norm_h))
            
            line = f"{cls_id} {x_center:.6f} {y_center:.6f} {norm_w:.6f} {norm_h:.6f}"
            yolo_lines.append(line)
        
        # 写入标签文件
        stem = Path(img_filename).stem
        label_file = labels_out / f"{stem}.txt"
        
        with open(label_file, 'w') as f:
            f.write('\n'.join(yolo_lines))
        
        # 复制或链接图像 (可选)
        if img_dir:
            src = Path(img_dir) / img_filename
            dst = images_out / img_filename
            if src.exists():
                import shutil
                shutil.copy2(src, dst)
        
        converted += 1
        
        if converted % 1000 == 0:
            print(f"⏳ 已处理 {converted}/{len(image_ids)} 张图像")
    
    # 生成data.yaml
    yaml_content = f"""path: {os.path.abspath(output_dir)}
train: images
val: images

nc: {len(categories)}
names: {{"""
    
    for idx, cat in enumerate(categories):
        yaml_content += f"\n  {idx}: {cat['name']}"
    
    yaml_content += "\n}"
    
    with open(Path(output_dir) / 'data.yaml', 'w') as f:
        f.write(yaml_content)
    
    print(f"\n✅ 转换完成!")
    print(f"   图像数量: {converted}")
    print(f"   类别数量: {len(categories)}")
    print(f"   输出目录: {output_dir}")
    print(f"   配置文件: {Path(output_dir) / 'data.yaml'}")


if __name__ == '__main__':
    coco_to_yolo(
        coco_json_path='annotations/instances_val2017.json',
        output_dir='coco_yolo_format/',
        img_dir='val2017/'
    )
```

## 📊 数据集划分策略

### 划分方法实现

```python
import os
import random
import shutil
from pathlib import Path
from collections import Counter


def split_dataset(dataset_root,
                  train_ratio=0.8,
                  val_ratio=0.1,
                  test_ratio=0.1,
                  seed=42,
                  copy_or_link='copy'):
    """
    数据集划分为训练/验证/测试集
    
    参数:
        dataset_root: 数据集根目录 (应包含images/和labels/)
        train_ratio: 训练集比例
        val_ratio: 验证集比例
        test_ratio: 测试集比例
        seed: 随机种子 (确保可复现)
        copy_or_link: 'copy'复制文件 or 'symlink'创建符号链接
    
    返回:
        统计信息字典
    """
    assert abs(train_ratio + val_ratio + test_ratio - 1.0) < 1e-6, \
        "比例之和必须等于1"
    
    random.seed(seed)
    
    images_dir = Path(dataset_root) / 'images'
    labels_dir = Path(dataset_root) / 'labels'
    
    # 获取所有图像文件
    all_images = sorted([f for f in images_dir.iterdir() 
                         if f.suffix.lower() in ['.jpg', '.jpeg', '.png', '.bmp']])
    
    print(f"📊 总计发现 {len(all_images)} 张图像")
    
    # 打乱顺序
    random.shuffle(all_images)
    
    # 计算各集合的数量
    n_total = len(all_images)
    n_train = int(n_total * train_ratio)
    n_val = int(n_total * val_ratio)
    n_test = n_total - n_train - n_val
    
    # 划分
    train_images = all_images[:n_train]
    val_images = all_images[n_train:n_train+n_val]
    test_images = all_images[n_train+n_val:]
    
    # 创建输出目录结构
    splits = {
        'train': train_images,
        'val': val_images,
        'test': test_images
    }
    
    stats = {'total': n_total}
    
    for split_name, image_list in splits.items():
        # 创建目录
        split_img_dir = Path(dataset_root) / 'images' / split_name
        split_lbl_dir = Path(dataset_root) / 'labels' / split_name
        split_img_dir.mkdir(exist_ok=True)
        split_lbl_dir.mkdir(exist_ok=True)
        
        # 移动/链接文件
        count = 0
        for img_path in image_list:
            # 处理图像
            dst_img = split_img_dir / img_path.name
            if copy_or_link == 'copy':
                shutil.copy2(img_path, dst_img)
            else:
                dst_img.symlink_to(img_path.resolve())
            
            # 处理对应的标签文件
            lbl_stem = img_path.stem
            lbl_extensions = ['.txt']
            
            for ext in lbl_extensions:
                src_lbl = labels_dir / (lbl_stem + ext)
                if src_lbl.exists():
                    dst_lbl = split_lbl_dir / src_lbl.name
                    if copy_or_link == 'copy':
                        shutil.copy2(src_lbl, dst_lbl)
                    else:
                        dst_lbl.symlink_to(src_lbl.resolve())
                    break
            
            count += 1
        
        stats[split_name] = count
        print(f"✅ {split_name}: {count} 张 ({count/n_total*100:.1f}%)")
    
    return stats


def analyze_dataset_distribution(labels_dir):
    """分析数据集类别分布"""
    from collections import Counter
    
    class_counts = Counter()
    total_objects = 0
    
    label_files = list(Path(labels_dir).glob('*.txt'))
    
    for label_file in label_files:
        with open(label_file, 'r') as f:
            lines = f.readlines()
        
        for line in lines:
            parts = line.strip().split()
            if len(parts) >= 1:
                class_id = int(parts[0])
                class_counts[class_id] += 1
                total_objects += 1
    
    print("\n📈 类别分布统计:")
    print("-" * 50)
    print(f"{'类别ID':<10} {'目标数量':<12} {'占比':<10}")
    print("-" * 50)
    
    for cls_id in sorted(class_counts.keys()):
        count = class_counts[cls_id]
        ratio = count / total_objects * 100
        print(f"{cls_id:<10} {count:<12} {ratio:<10.2f}%")
    
    print("-" * 50)
    print(f"{'总计':<10} {total_objects:<12} {'100%':<10}")
    
    return dict(class_counts)


if __name__ == '__main__':
    # 示例：划分自定义数据集
    stats = split_dataset(
        dataset_root='my_custom_dataset/',
        train_ratio=0.8,
        val_ratio=0.15,
        test_ratio=0.05,
        seed=42
    )
    
    # 分析训练集分布
    analyze_dataset_distribution('my_custom_dataset/labels/train/')
```

## 🎨 数据增强策略

### 自定义增强管道

```python
import albumentations as A
from albumentations.pytorch import ToTensorV2
import cv2
import numpy as np
from pathlib import Path
import yaml


class YOLOAugmentationPipeline:
    """YOLO专用数据增强管道"""
    
    def __init__(self, 
                 img_size=640,
                 augment=True,
                 mosaic_prob=1.0,
                 mixup_prob=0.0,
                 hsv_aug=True):
        self.img_size = img_size
        self.augment = augment
        
        if augment:
            # Albumentations增强
            self.transform = A.Compose([
                # 几何变换
                A.RandomResizedCrop(img_size, img_size, scale=(0.8, 1.0)),
                A.HorizontalFlip(p=0.5),
                A.VerticalFlip(p=0.05),
                A.ShiftScaleRotate(shift_limit=0.0625, scale_limit=0.1, 
                                   rotate_limit=15, p=0.3),
                
                # 颜色空间变换
                A.OneOf([
                    A.RandomBrightnessContrast(brightness_limit=0.2, contrast_limit=0.2),
                    A.CLAHE(clip_limit=2),
                    A.HueSaturationValue(hue_shift_limit=20, sat_shift_limit=30, 
                                          val_shift_limit=20),
                ], p=0.3),
                
                # 噪声和模糊
                A.OneOf([
                    A.GaussNoise(var_limit=(10, 50), p=0.5),
                    A.GaussianBlur(blur_limit=(3, 7), p=0.5),
                ], p=0.2),
                
                # 归一化
                A.Normalize(mean=[0.485, 0.456, 0.406], 
                           std=[0.229, 0.224, 0.225]),
                ToTensorV2(),
            ],
            bbox_params=A.BboxParams(format='yolov8', 
                                   label_fields=['class_labels'],
                                   min_visibility=0.3))
        else:
            # 仅调整大小和归一化
            self.transform = A.Compose([
                A.Resize(img_size, img_size),
                A.Normalize(mean=[0.485, 0.456, 0.406], 
                           std=[0.229, 0.224, 0.225]),
                ToTensorV2(),
            ],
            bbox_params=A.BboxParams(format='yolov8',
                                   label_fields=['class_labels']))
    
    def __call__(self, image, bboxes, class_labels):
        """
        应用增强
        
        参数:
            image: BGR图像 (H, W, C)
            bboxes: YOLO格式的边界框列表 [[xc, yc, w, h], ...]
            class_labels: 类别ID列表 [int, int, ...]
        
        返回:
            augmented: 字典包含 'image', 'bboxes', 'class_labels'
        """
        if not self.augment:
            # 不增强时只做resize
            resized = cv2.resize(image, (self.img_size, self.img_size))
            return {
                'image': resized,
                'bboxes': bboxes,
                'class_labels': class_labels
            }
        
        # 应用albumentations增强
        augmented = self.transform(
            image=image, 
            bboxes=bboxes, 
            class_labels=class_labels
        )
        
        return augmented


# Mosaic增强实现 (类似Ultralytics内置)
def create_mosaic_image(images, labels, img_size=640):
    """
    Mosaic增强: 将4张图片拼接成一张
    
    参数:
        images: 4张图像的列表 [img1, img2, img3, img4]
        labels: 对应的标签列表 [lbls1, lbls2, lbls3, lbls4]
        img_size: 输出图像尺寸
    
    返回:
        mosaic_img: 拼接后的图像
        mosaic_labels: 合并后的标签 (已重新计算坐标)
    """
    assert len(images) == 4, "Mosaic需要恰好4张图像"
    
    # 创建画布
    mosaic_img = np.full((img_size * 2, img_size * 2, 3), 114, dtype=np.uint8)
    
    # 定义4个子区域的位置
    centers = [(img_size // 2, img_size // 2),
               (img_size + img_size // 2, img_size // 2),
               (img_size // 2, img_size + img_size // 2),
               (img_size + img_size // 2, img_size + img_size // 2)]
    
    mosaic_labels = []
    
    for i, (img, lbls, center) in enumerate(zip(images, labels, centers)):
        h, w = img.shape[:2]
        
        # 缩放图像以适应子区域
        scale = min(img_size / h, img_size / w)
        new_h, new_w = int(h * scale), int(w * scale)
        img_resized = cv2.resize(img, (new_w, new_h))
        
        # 计算放置位置 (随机偏移)
        cx, cy = center
        offset_x = np.random.randint(-new_w//4, new_w//4)
        offset_y = np.random.randint(-new_h//4, new_h//4)
        
        start_x = cx - new_w // 2 + offset_x
        start_y = cy - new_h // 2 + offset_y
        
        # 粘贴到画布
        end_x = start_x + new_w
        end_y = start_y + new_h
        
        # 边界检查
        sx, sy = max(0, start_x), max(0, start_y)
        ex, ey = min(img_size*2, end_x), min(img_size*2, end_y)
        ix, iy = sx - start_x, sy - start_y
        jx, jy = ex - start_x, ey - start_y
        
        if ex > sx and ey > sy and jx > ix and jy > iy:
            mosaic_img[sy:ey, sx:ex] = img_resized[iy:jy, ix:jx]
        
        # 调整标签坐标
        for lbl in lbls:
            cls_id, xc, yc, bw, bh = lbl
            
            # 新坐标 (相对于整个mosaic图像)
            new_xc = (xc * w * scale + start_x) / (img_size * 2)
            new_yc = (yc * h * scale + start_y) / (img_size * 2)
            new_bw = (bw * w * scale) / (img_size * 2)
            new_bh = (bh * h * scale) / (img_size * 2)
            
            # 过滤超出边界的框
            if (0 < new_xc < 1 and 0 < new_yc < 1 and 
                0 < new_xc + new_bw/2 < 1 and 0 < new_yc + new_bh/2 < 1):
                mosaic_labels.append((cls_id, new_xc, new_yc, new_bw, new_bh))
    
    # 随机裁剪到目标尺寸
    crop_x = np.random.randint(0, img_size)
    crop_y = np.random.randint(0, img_size)
    
    final_img = mosaic_img[crop_y:crop_y+img_size, crop_x:crop_x+img_size]
    
    # 再次调整标签坐标
    final_labels = []
    for lbl in mosaic_labels:
        cls_id, xc, yc, bw, bh = lbl
        new_xc = (xc * 2 - crop_x / img_size)
        new_yc = (yc * 2 - crop_y / img_size)
        
        if 0 < new_xc < 1 and 0 < new_yc < 1:
            final_labels.append((cls_id, new_xc, new_yc, bw, bh))
    
    return final_img, final_labels

def validate_yolo_dataset(dataset_yaml_path):
    """
    全面验证YOLO数据集的质量和完整性
    
    检查项:
    1. 文件完整性 (图像和标签一一对应)
    2. 标注格式正确性
    3. 坐标范围有效性
    4. 类别一致性
    5. 数据分布均衡性
    """
    import yaml
    from pathlib import Path
    
    print("=" * 60)
    print("🔍 YOLO数据集质量验证")
    print("=" * 60)
    
    # 加载配置
    with open(dataset_yaml_path, 'r') as f:
        config = yaml.safe_load(f)
    
    base_path = Path(config['path'])
    num_classes = config['nc']
    class_names = config.get('names', {})
    
    issues = []
    statistics = {
        'total_images': 0,
        'total_annotations': 0,
        'empty_labels': 0,
        'missing_labels': 0,
        'invalid_coords': 0,
        'class_distribution': {}
    }
    
    # 检查每个split
    for split in ['train', 'val']:
        if split not in config:
            continue
        
        img_dir = base_path / 'images' / split
        lbl_dir = base_path / 'labels' / split
        
        if not img_dir.exists():
            issues.append(f"❌ 图像目录不存在: {img_dir}")
            continue
        
        image_files = list(img_dir.glob('*.*'))
        
        for img_file in image_files:
            if img_file.suffix.lower() not in ['.jpg', '.png', '.jpeg', '.bmp']:
                continue
            
            statistics['total_images'] += 1
            stem = img_file.stem
            lbl_file = lbl_dir / f"{stem}.txt"
            
            # 检查标签文件是否存在
            if not lbl_file.exists():
                statistics['missing_labels'] += 1
                issues.append(f"⚠️  缺少标签: {img_file.name}")
                continue
            
            # 解析标签
            with open(lbl_file, 'r') as f:
                lines = f.readlines()
            
            if len(lines) == 0:
                statistics['empty_labels'] += 1
                continue
            
            for line_num, line in enumerate(lines, 1):
                parts = line.strip().split()
                
                if len(parts) < 5:
                    issues.append(f"❌ 格式错误 {lbl_file}:{line_num}")
                    continue
                
                try:
                    cls_id = int(parts[0])
                    xc, yc, w, h = map(float, parts[1:5])
                    
                    # 验证类别ID
                    if cls_id < 0 or cls_id >= num_classes:
                        issues.append(f"❌ 无效类别ID {cls_id} in {lbl_file}:{line_num}")
                    
                    # 验证坐标范围
                    if not (0 <= xc <= 1 and 0 <= yc <= 1 and 
                            0 < w <= 1 and 0 < h <= 1):
                        statistics['invalid_coords'] += 1
                        issues.append(f"⚠️  坐标越界 {lbl_file}:{line_num}")
                    
                    # 统计类别分布
                    statistics['class_distribution'][cls_id] = \
                        statistics['class_distribution'].get(cls_id, 0) + 1
                    
                    statistics['total_annotations'] += 1
                    
                except ValueError:
                    issues.append(f"❌ 数值解析错误 {lbl_file}:{line_num}")
    
    # 输出结果
    print(f"\n📊 数据集统计:")
    print(f"   总图像数: {statistics['total_images']}")
    print(f"   总标注数: {statistics['total_annotations']}")
    print(f"   空标签文件: {statistics['empty_labels']}")
    print(f"   缺失标签: {statistics['missing_labels']}")
    print(f"   无效坐标: {statistics['invalid_coords']}")
    
    print(f"\n📈 类别分布:")
    for cls_id in sorted(statistics['class_distribution'].keys()):
        count = statistics['class_distribution'][cls_id]
        name = class_names.get(cls_id, f'class_{cls_id}')
        ratio = count / statistics['total_annotations'] * 100
        bar = '█' * int(ratio / 5)
        print(f"   {cls_id:>3}. {name:<20} {count:>6} ({ratio:>5.1f}%) {bar}")
    
    if issues:
        print(f"\n⚠️  发现 {len(issues)} 个问题:")
        for issue in issues[:20]:  # 只显示前20个
            print(f"   {issue}")
        if len(issues) > 20:
            print(f"   ... 还有 {len(issues)-20} 个问题")
    else:
        print("\n✅ 数据集验证通过! 无明显问题")
    
    return len(issues) == 0


if __name__ == '__main__':
    validate_yolo_dataset('my_dataset/data.yaml')
```

## 💡 最佳实践总结

### ✅ 推荐流程

1. **收集数据**: 确保数据多样性和代表性
2. **选择工具**: LabelImg (简单) / CVAT (协作) / LabelMe (分割)
3. **质量控制**: 使用验证脚本检查标注质量
4. **格式转换**: 使用提供的脚本转换为YOLO格式
5. **合理划分**: 通常80%训练 / 10%验证 / 10%测试
6. **适度增强**: 根据数据量调整增强强度
7. **文档记录**: 保存数据集版本和处理日志

### ⚠️ 注意事项

- ⚠️ **保持一致**: 训练和验证使用相同的预处理
- ⚠️ **避免泄露**: 同一场景的图像不要分散在不同split中
- ⚠️ **平衡类别**: 处理类别不平衡问题
- ⚠️ **备份原始数据**: 保留未处理的原始标注
