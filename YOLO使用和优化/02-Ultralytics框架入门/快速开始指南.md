## 🎯 第二步：第一个检测程序

### 方法一：Python API（推荐用于开发）

创建 `first_detection.py`:

```python
from ultralytics import YOLO


def main():
    """
    你的第一个YOLOv8检测程序
    
    功能: 加载预训练模型，对图片进行目标检测，并可视化结果
    预计耗时: 首次运行会下载模型 (~6MB)，后续推理 <100ms (GPU)
    """
    
    print("=" * 60)
    print("🎯 YOLOv8 第一个检测程序")
    print("=" * 60)
    
    # ===== 步骤1: 加载预训练模型 =====
    # 可选模型:
    # - yolov8n.pt (nano): 最快，精度较低，适合实时应用
    # - yolov8s.pt (small): 平衡速度与精度
    # - yolov8m.pt (medium): 更高精度
    # - yolov8l.pt (large): 高精度
    # - yolov8x.pt (xlarge): 最高精度，最慢
    
    model = YOLO('yolov8n.pt')  # 使用 nano 版本，速度快
    print("✅ 模型加载成功")
    
    # ===== 步骤2: 对图片进行推理 =====
    # 支持的输入格式:
    # - 图片路径 (str)
    # - PIL Image对象
    # - numpy数组 (HWC或CHW格式)
    # - 视频路径 (.mp4, .avi等)
    # - 摄像头索引 (0, 1, ...)
    # - 文件夹路径 (批量处理)
    # - URL地址
    
    image_source = 'https://ultralytics.com/images/bus.jpg'  # 使用在线示例图片
    
    print(f"🔍 正在检测图片: {image_source}")
    results = model(image_source)  # 返回 Results 对象列表
    print(f"✅ 检测完成，发现 {len(results)} 张图片的结果")
    
    # ===== 步骤3: 解析检测结果 =====
    for i, result in enumerate(results):
        print(f"\n--- 图片 {i+1} 检测结果 ---")
        
        # 获取检测到的目标数量
        num_detections = len(result.boxes)
        print(f"📦 检测到 {num_detections} 个目标")
        
        # 遍历每个检测框
        if num_detections > 0:
            boxes = result.boxes  # Boxes 对象
            
            for j in range(num_detections):
                # 获取边界框坐标 [x1, y1, x2, y2]
                xyxy = boxes.xyxy[j].cpu().numpy()
                
                # 获取置信度分数
                confidence = boxes.conf[j].cpu().numpy()
                
                # 获取类别ID和名称
                class_id = int(boxes.cls[j].cpu().numpy())
                class_name = result.names[class_id]
                
                print(f"  目标 {j+1}: {class_name} "
                      f"(置信度: {confidence:.2%}, "
                      f"位置: [{xyxy[0]:.0f}, {xyxy[1]:.0f}, {xyxy[2]:.0f}, {xyxy[3]:.0f}])")
    
    # ===== 步骤4: 保存可视化结果 =====
    # 保存带标注框的图片
    output_path = 'results/detection_result.jpg'
    results[0].save(filename=output_path)
    print(f"\n💾 结果已保存到: {output_path}")
    
    # 或者使用 plot() 方法获取 numpy 数组进行自定义处理
    plotted_img = results[0].plot()
    import cv2
    cv2.imwrite('results/custom_output.jpg', plotted_img)
    print(f"💾 自定义结果已保存到: results/custom_output.jpg")
    
    print("\n" + "=" * 60)
    print("🎉 恭喜！你已经成功运行了第一个YOLO程序！")
    print("=" * 60)


if __name__ == '__main__':
    import os
    os.makedirs('results', exist_ok=True)
    main()
```

```python
import cv2
from pathlib import Path
from ultralytics import YOLO


class YOLODetector:
    """YOLO检测器封装类"""
    
    def __init__(self, model_name='yolov8n.pt', conf_threshold=0.25):
        """
        初始化检测器
        
        参数:
            model_name: 模型文件名或路径
            conf_threshold: 置信度阈值
        """
        self.model = YOLO(model_name)
        self.conf = conf_threshold
        
    def detect_image_file(self, image_path):
        """检测图片文件"""
        results = self.model(
            source=image_path,
            conf=self.conf,
            verbose=False
        )
        return results
    
    def detect_image_array(self, image_array):
        """检测numpy数组格式的图片"""
        # 支持HWC和BGR/RGB格式
        results = self.model(
            source=image_array,
            conf=self.conf,
            verbose=False
        )
        return results
    
    def detect_video(self, video_path, output_path=None, show=False):
        """
        检测视频并保存结果
        
        参数:
            video_path: 输入视频路径
            output_path: 输出视频路径 (None则自动生成)
            show: 是否显示窗口
        """
        cap = cv2.VideoCapture(video_path)
        
        if not cap.isOpened():
            raise ValueError(f"无法打开视频: {video_path}")
        
        # 获取视频信息
        fps = int(cap.get(cv2.CAP_PROP_FPS))
        width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
        height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
        total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
        
        print(f"📹 视频信息: {width}x{height} @ {fps} FPS, 共{total_frames}帧")
        
        # 设置输出视频编码器
        if output_path is None:
            output_path = str(Path(video_path).with_suffix('.detected.mp4'))
        
        fourcc = cv2.VideoWriter_fourcc(*'mp4v')
        out = cv2.VideoWriter(output_path, fourcc, fps, (width, height))
        
        frame_count = 0
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            
            frame_count += 1
            
            # 检测当前帧
            results = self.detect_image_array(frame)
            
            # 绘制检测结果
            annotated_frame = results[0].plot()
            
            # 写入输出视频
            out.write(annotated_frame)
            
            # 显示进度
            if frame_count % 30 == 0:
                print(f"⏳ 已处理 {frame_count}/{total_frames} 帧 ({frame_count/total_frames*100:.1f}%)")
            
            # 显示窗口
            if show:
                cv2.imshow('YOLO Detection', annotated_frame)
                if cv2.waitKey(1) & 0xFF == ord('q'):
                    break
        
        # 释放资源
        cap.release()
        out.release()
        cv2.destroyAllWindows()
        
        print(f"✅ 视频处理完成！保存至: {output_path}")
        return output_path
    
    def detect_webcam(self, camera_index=0):
        """
        实时摄像头检测
        
        参数:
            camera_index: 摄像头设备索引
        """
        cap = cv2.VideoCapture(camera_index)
        
        if not cap.isOpened():
            raise ValueError(f"无法打开摄像头: {camera_index}")
        
        print("📷 摄像头实时检测中... 按 'q' 键退出")
        
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            
            # 检测
            results = self.detect_image_array(frame)
            
            # 绘制结果
            annotated_frame = results[0].plot()
            
            # 显示
            cv2.imshow('Webcam Detection', annotated_frame)
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        cap.release()
        cv2.destroyAllWindows()


# 使用示例
if __name__ == '__main__':
    detector = YOLODetector(
        model_name='yolov8n.pt',
        conf_threshold=0.5  # 提高置信度减少误检
    )
    
    # 示例1: 检测图片
```

```python
from ultralytics import YOLO
import numpy as np


def analyze_results_detailed():
    """详细分析YOLO返回的结果结构"""
    
    model = YOLO('yolov8n.pt')
    results = model('https://ultralytics.com/images/zidane.jpg', verbose=False)
    
    result = results[0]  # 取第一张图的结果
    
    print("=" * 70)
    print("📊 Results 对象详细属性")
    print("=" * 70)
    
    # ===== 基本信息 =====
    print("\n【基本信息】")
    print(f"原始图像形状: {result.orig_shape}")  # (720, 1280)
    print(f"输入图像形状: {result.shape}")       # (3, 640, 640)
    
    # ===== 类别信息 =====
    print("\n【类别信息】")
    print(f"所有类别名称: {result.names}")
    # {'0': 'person', '1': 'bicycle', '2': 'car', ...}
    
    # ===== 边界框信息 (Boxes) =====
    print("\n【边界框信息】")
    boxes = result.boxes
    
    print(f"检测到的目标数量: {len(boxes)}")
    
    if len(boxes) > 0:
        # 1. 边界框坐标 (xyxy格式: x1, y1, x2, y2)
        xyxy = boxes.xyxy  # Tensor [N, 4]
        print(f"\n边界框坐标 (xyxy):")
        print(xyxy)
        
        # 2. 边界框坐标 (xywh格式: x_center, y_center, width, height)
        xywh = boxes.xywh  # Tensor [N, 4]
        print(f"\n边界框坐标 (xywh):")
        print(xywh)
        
        # 3. 置信度分数
        confidences = boxes.conf  # Tensor [N]
        print(f"\n置信度分数:")
        for i, conf in enumerate(confidences):
            print(f"  目标{i}: {conf.item():.4f} ({conf.item()*100:.2f}%)")
        
        # 4. 类别预测
        classes = boxes.cls  # Tensor [N]
        print(f"\n类别预测:")
        for i, cls_id in enumerate(classes):
            cls_name = result.names[int(cls_id.item())]
            print(f"  目标{i}: 类别ID={int(cls_id.item())}, 名称={cls_name}")
        
        # 5. 转换为numpy方便后续处理
        boxes_np = boxes.cpu().numpy()
        print(f"\n转换为NumPy数组后的形状: {boxes_np.shape}")
        
        # ===== 提取特定信息的方法 =====
        print("\n【数据提取示例】")
        
        # 方法A: 获取所有目标的详细信息列表
        detections_list = []
        for box in boxes:
            det = {
                'bbox_xyxy': box.xyxy[0].cpu().tolist(),      # [x1, y1, x2, y2]
                'bbox_xywh': box.xywh[0].cpu().tolist(),      # [cx, cy, w, h]
                'confidence': box.conf[0].cpu().item(),        # 置信度
                'class_id': int(box.cls[0].cpu().item()),      # 类别ID
                'class_name': result.names[int(box.cls[0].cpu().item())]  # 类别名
            }
            detections_list.append(det)
            
            print(f"\n目标详情:")
            print(f"  位置 (xyxy): {det['bbox_xyxy']}")
            print(f"  位置 (xywh): {det['bbox_xywh']}")
            print(f"  置信度: {det['confidence']:.2%}")
            print(f"  类别: {det['class_name']} (ID: {det['class_id']})")
        
        # 方法B: 过滤特定类别的目标
        person_boxes = boxes[boxes.cls == 0]  # 只保留person类别
        print(f"\n只保留person类别的数量: {len(person_boxes)}")
        
        # 方法C: 过滤高置信度的目标
        high_conf_boxes = boxes[boxes.conf > 0.8]
        print(f"置信度>80%的目标数量: {len(high_conf_boxes)}")
        
        # 方法D: 按置信度排序
        sorted_indices = boxes.conf.argsort(descending=True)
        sorted_boxes = boxes[sorted_indices]
        print(f"\n按置信度排序后:")
        for i, box in enumerate(sorted_boxes[:3]):  # 只显示前3个
            cls_name = result.names[int(box.cls[0])]
            print(f"  Top-{i+1}: {cls_name} ({box.conf[0]:.2%})")


if __name__ == '__main__':
    analyze_results_detailed()
```

## 🖼️ 第五步：自定义可视化

```python
import cv2
import numpy as np
from ultralytics import YOLO


def custom_visualization():
    """自定义检测结果的可视化样式"""
    
    model = YOLO('yolov8n.pt')
    results = model('https://ultralytics.com/images/bus.jpg', verbose=False)
    result = results[0]
    
    # 获取原始图像
    orig_img = result.orig_img.copy()
    
    # 定义颜色映射 (BGR格式)
    COLORS = {
        'person': (255, 0, 0),      # 红色
        'bicycle': (0, 255, 0),     # 绿色
        'car': (0, 0, 255),         # 蓝色
        'motorcycle': (255, 255, 0), # 青色
        'bus': (255, 0, 255),       # 紫色
        'truck': (0, 255, 255),     # 黄色
    }
    
    # 默认颜色 (对于未定义的类别)
    DEFAULT_COLOR = (128, 128, 128)  # 灰色
    
    # 设置字体
    font = cv2.FONT_HERSHEY_SIMPLEX
    font_scale = 0.7
    thickness = 2
    
    boxes = result.boxes
    
    for i, box in enumerate(boxes):
        # 获取坐标 (转换为整数)
        x1, y1, x2, y2 = map(int, box.xyxy[0].cpu().numpy())
        
        # 获取类别和置信度
        class_id = int(box.cls[0].cpu().numpy())
        class_name = result.names[class_id]
        confidence = box.conf[0].cpu().numpy()
        
        # 选择颜色
        color = COLORS.get(class_name, DEFAULT_COLOR)
        
        # ===== 绘制边界框 =====
        # 外边框 (粗线)
        cv2.rectangle(orig_img, (x1, y1), (x2, y2), color, thickness)
        
        # 内边框 (细线，形成双线效果)
        cv2.rectangle(orig_img, (x1+2, y1+2), (x2-2, y2-2), color, 1)
        
        # ===== 绘制标签背景 =====
        label_text = f"{class_name} {confidence:.0%}"
        (text_width, text_height), _ = cv2.getTextSize(label_text, font, font_scale, thickness)
        
        # 标签背景矩形
        label_y1 = max(y1 - text_height - 10, 0)
        cv2.rectangle(orig_img, 
                     (x1, label_y1), 
                     (x1 + text_width + 10, y1),
                     color, 
                     -1)  # 填充
        
        # ===== 绘制标签文字 =====
        cv2.putText(orig_img, 
                   label_text,
                   (x1 + 5, y1 - 5),
                   font, 
                   font_scale, 
                   (255, 255, 255),  # 白色文字
                   thickness)
        
        # ===== 可选: 绘制中心点 =====
        cx = (x1 + x2) // 2
        cy = (y1 + y2) // 2
        cv2.circle(orig_img, (cx, cy), 4, color, -1)
        
        # ===== 可选: 绘制角点标记 =====
        corner_size = 10
        corners = [
            [(x1, y1), (x1 + corner_size, y1), (x1, y1 + corner_size)],           # 左上
            [(x2, y1), (x2 - corner_size, y1), (x2, y1 + corner_size)],           # 右上
            [(x1, y2), (x1 + corner_size, y2), (x1, y2 - corner_size)],           # 左下
            [(x2, y2), (x2 - corner_size, y2), (x2, y2 - corner_size)]            # 右下
        ]
        for corner in corners:
            pts = np.array(corner, dtype=np.int32)
            cv2.polylines(orig_img, [pts], False, color, 2)
    
    # 添加标题
    title_text = f"Custom YOLO Detection | Objects: {len(boxes)}"
    cv2.putText(orig_img, title_text, (20, 40), font, 1, (255, 255, 255), 3)
    cv2.putText(orig_img, title_text, (20, 40), font, 1, (0, 0, 0), 2)
    
    # 保存结果
    cv2.imwrite('custom_visualization.jpg', orig_img)
    print("✅ 自定义可视化结果已保存: custom_visualization.jpg")
    
    return orig_img


if __name__ == '__main__':
    custom_visualization()
```

## ⚡ 性能对比测试

### 不同模型的性能基准

```python
import time
from ultralytics import YOLO
import matplotlib.pyplot as plt


def benchmark_models(image_source='https://ultralytics.com/images/bus.jpg'):
    """
    测试不同YOLOv8模型的性能
    
    返回:
        benchmark_data: 包含各模型性能指标的字典
    """
    
    models_to_test = {
        'yolov8n': 'yolov8n.pt',   # nano: 3.2M参数
        'yolov8s': 'yolov8s.pt',   # small: 11.2M参数
        'yolov8m': 'yolov8m.pt',   # medium: 25.9M参数
        # 'yolov8l': 'yolov8l.pt',  # large: 43.7M参数 (可选)
        # 'yolov8x': 'yolov8x.pt',  # xlarge: 68.2M参数 (可选)
    }
    
    benchmark_data = {}
    
    print("=" * 70)
    print("🏃 YOLOv8 模型性能基准测试")
    print("=" * 70)
    
    for model_name, model_path in models_to_test.items():
        print(f"\n{'='*50}")
        print(f"🔍 测试模型: {model_name}")
        print(f"{'='*50}")
        
        # 加载模型
        model = YOLO(model_path)
        
        # 预热 (让GPU达到稳定状态)
        _ = model(image_source, verbose=False)
        
        # 多次测试取平均值
        num_runs = 10
        times = []
        num_objects_list = []
        
        for i in range(num_runs):
            start_time = time.perf_counter()
            results = model(image_source, verbose=False)
            end_time = time.perf_counter()
            
            elapsed_ms = (end_time - start_time) * 1000
            times.append(elapsed_ms)
            
            num_objects = len(results[0].boxes)
            num_objects_list.append(num_objects)
        
        # 计算统计指标
        avg_time = sum(times) / len(times)
        min_time = min(times)
        max_time = max(times)
        std_time = (sum((t - avg_time)**2 for t in times) / len(times)) ** 0.5
        fps = 1000 / avg_time
        
        avg_objects = sum(num_objects_list) / len(num_objects_list)
        
        # 存储结果
        benchmark_data[model_name] = {
            'avg_time_ms': avg_time,
            'min_time_ms': min_time,
            'max_time_ms': max_time,
            'std_time_ms': std_time,
            'fps': fps,
            'avg_objects': avg_objects
        }
        
        print(f"⏱️  平均推理时间: {avg_time:.2f} ms")
        print(f"📊 时间范围: {min_time:.2f} - {max_time:.2f} ms")
        print(f"📈 标准差: {std_time:.2f} ms")
        print(f"🎯 平均FPS: {fps:.1f}")
        print(f"📦 平均检测目标数: {avg_objects:.1f}")
    
    # 打印汇总表格
    print("\n" + "=" * 70)
    print("📋 性能汇总表")
    print("=" * 70)
    print(f"{'模型':<10} {'平均时间(ms)':<15} {'FPS':<10} {'检测数':<10}")
    print("-" * 50)
    
    for model_name, data in benchmark_data.items():
        print(f"{model_name:<10} {data['avg_time_ms']:<15.2f} {data['fps']:<10.1f} {data['avg_objects']:<10.1f}")
    
    return benchmark_data
```
